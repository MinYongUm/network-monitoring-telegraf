# -----------------------------------------------------------------------------
# HTTP 클라이언트
#
# requests: 아래 두 용도로 사용
#   [aci_collector.py]
#   - POST /api/aaaLogin.json   : APIC 토큰 인증
#   - GET  /api/class/*.json    : 메트릭 수집 (faultInst, fabricHealthTotal 등)
#   [slack_notify.py]
#   - POST Slack Incoming Webhook URL : 알림 메시지 전송
#   - SSL 검증 비활성화(verify=False) 옵션 사용 시 urllib3 경고 억제 필요
# -----------------------------------------------------------------------------
requests==2.31.0

# urllib3 는 requests 의 의존성이지만 버전 명시로 호환성 보장
urllib3==2.1.0


# -----------------------------------------------------------------------------
# SSL 경고 억제
#
# urllib3 의 InsecureRequestWarning 억제에 사용
# ACI APIC 의 자체 서명 인증서(Self-signed certificate) 환경에서 필요
# -----------------------------------------------------------------------------
certifi==2024.2.2


# -----------------------------------------------------------------------------
# 환경변수 관리
#
# python-dotenv: .env 파일에서 환경변수 로드
#   - 로컬 개발 환경에서 직접 스크립트 실행 시 사용
#   - docker compose 환경에서는 environment 로 주입되므로 선택적 사용
# -----------------------------------------------------------------------------
python-dotenv==1.0.1


# -----------------------------------------------------------------------------
# InfluxDB 클라이언트 (v0.2.0 추가)
#
# influxdb-client: aci_collector.py 에서 InfluxDB HTTP API 직접 write 에 사용
#   - v0.1.x: Telegraf exec 플러그인 → stdout line protocol → Telegraf → InfluxDB
#   - v0.2.0: aci-collector 루프 → InfluxDB 직접 write (C안 전환)
#
# 주요 사용:
#   - InfluxDBClient            : 연결 및 세션 관리
#   - write_api(SYNCHRONOUS)    : 동기 write (에러 즉시 감지)
#   - record=list[str]          : line protocol 문자열 목록 직접 전달
# -----------------------------------------------------------------------------
influxdb-client==1.48.0


# -----------------------------------------------------------------------------
# Webhook 수신 서버
#
# flask: alerting/slack_notify.py 의 Grafana Webhook 수신 서버
#   - POST /alert  : Grafana Alert Webhook 수신 엔드포인트
#   - GET  /health : 헬스체크 엔드포인트
#
# gunicorn: Flask 운영 환경 WSGI 서버
#   - 개발 환경: python slack_notify.py (Flask 내장 서버)
#   - 운영 환경: gunicorn -w 2 -b 0.0.0.0:5001 slack_notify:app
#   Stage 1 (POC) 에서는 Flask 내장 서버로 충분하나
#   안정적인 Webhook 수신을 위해 gunicorn 사용 권장
# -----------------------------------------------------------------------------
flask==3.0.3
gunicorn==22.0.0
